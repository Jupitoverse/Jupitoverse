import requests
import json

url = "https://oso.orion.comcast.com/dop/security/rest/loginservice/login"

payload = json.dumps({
  "userID": "nap-admin-user",
  "password": "Nap-admin-user1"
})
headers = {
  'Content-Type': 'application/json'
}

response = requests.request("POST", url, headers=headers, data=payload)

print(response.text)
